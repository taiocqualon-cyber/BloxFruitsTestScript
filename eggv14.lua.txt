function GetEverySingleEgg()
    local EggsFound = {}
    -- Quét toàn bộ Descendants để tìm trong các Folder ẩn (nil parent)
    for _, v in pairs(game:GetDescendants()) do
        if v:IsA("BasePart") and v:FindFirstChild("TouchInterest") then
            -- 1. Lọc theo kích thước trứng trong ảnh bạn gửi (Nhỏ và gọn)
            local s = v.Size
            if (s.X > 1 and s.X < 6) and (s.Y > 1 and s.Y < 8) then
                -- 2. Kiểm tra tên: Phải chứa các từ khóa sự kiện bạn đã liệt kê
                local n = v.Name:lower()
                if n:find("egg") or n:find("indra") or n:find("rocket") or n:find("moon") or n:find("hacker") then
                    table.insert(EggsFound, v)
                end
            end
        end
    end
    return EggsFound
end
