-- LocalScript: VIP Luck Tester
local Players = game:GetService("Players")
local player = Players.LocalPlayer
local Workspace = game:GetService("Workspace")

-- GUI
local ScreenGui = Instance.new("ScreenGui", player:WaitForChild("PlayerGui"))
local Frame = Instance.new("Frame", ScreenGui)
Frame.Size = UDim2.new(0, 400, 0, 200)
Frame.Position = UDim2.new(0.5, -200, 0.5, -100)
Frame.BackgroundColor3 = Color3.fromRGB(25,25,25)

local TextBox = Instance.new("TextBox", Frame)
TextBox.Size = UDim2.new(1, -20, 0, 50)
TextBox.Position = UDim2.new(0, 10, 0, 10)
TextBox.PlaceholderText = "Nhập mã lệnh thử nghiệm"

local ExecuteBtn = Instance.new("TextButton", Frame)
ExecuteBtn.Size = UDim2.new(1, -20, 0, 50)
ExecuteBtn.Position = UDim2.new(0, 10, 0, 70)
ExecuteBtn.Text = "Thực hiện lệnh"

local OutputLabel = Instance.new("TextLabel", Frame)
OutputLabel.Size = UDim2.new(1, -20, 0, 50)
OutputLabel.Position = UDim2.new(0, 10, 0, 130)
OutputLabel.Text = "Kết quả thử nghiệm"
OutputLabel.TextScaled = true
OutputLabel.BackgroundTransparency = 1

-- Logic: Tăng may mắn local
ExecuteBtn.MouseButton1Click:Connect(function()
	local cmd = TextBox.Text
	if cmd:lower() == "maymanmax" then
		OutputLabel.Text = "Tỷ lệ drop local thử nghiệm: 100%"
	elseif cmd:lower() == "mayman50" then
		OutputLabel.Text = "Tỷ lệ drop local thử nghiệm: 50%"
	else
		OutputLabel.Text = "Lệnh không hợp lệ"
	end
end)
