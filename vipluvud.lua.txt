-- LocalScript: VIP Blox Fruits Hub (LocalTest)
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local TweenService = game:GetService("TweenService")
local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local rootPart = character:WaitForChild("HumanoidRootPart")
local humanoid = character:WaitForChild("Humanoid")

-- ================== GUI ==================
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "VIPBloxHubGUI"
ScreenGui.Parent = player:WaitForChild("PlayerGui")

local HubButton = Instance.new("TextButton")
HubButton.Size = UDim2.new(0,100,0,50)
HubButton.Position = UDim2.new(0,20,0,100)
HubButton.Text = "VIP Hub"
HubButton.BackgroundColor3 = Color3.fromRGB(50,50,50)
HubButton.TextColor3 = Color3.fromRGB(255,255,255)
HubButton.Parent = ScreenGui

local MenuFrame = Instance.new("Frame")
MenuFrame.Size = UDim2.new(0,300,0,400)
MenuFrame.Position = UDim2.new(0,140,0,100)
MenuFrame.BackgroundColor3 = Color3.fromRGB(30,30,30)
MenuFrame.Visible = false
MenuFrame.Parent = ScreenGui

local title = Instance.new("TextLabel")
title.Size = UDim2.new(1,0,0,50)
title.Text = "VIP BloxHub"
title.TextScaled = true
title.BackgroundTransparency = 1
title.TextColor3 = Color3.fromRGB(255,255,255)
title.Parent = MenuFrame

-- ================== Toggle Buttons ==================
local function createButton(name, parent, callback)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(1,-20,0,50)
    btn.Position = UDim2.new(0,10,0,50*(#parent:GetChildren()))
    btn.BackgroundColor3 = Color3.fromRGB(60,60,60)
    btn.TextColor3 = Color3.fromRGB(255,255,255)
    btn.Text = name
    btn.Parent = parent
    btn.MouseButton1Click:Connect(callback)
end

-- Toggle biến
local autoFarm = false
local autoLevel = false
local flyEnabled = false

createButton("Auto Farm Quái", MenuFrame, function()
    autoFarm = not autoFarm
    print("Auto Farm Quái:", autoFarm)
end)

createButton("Auto Level", MenuFrame, function()
    autoLevel = not autoLevel
    print("Auto Level:", autoLevel)
end)

createButton("Bay / Fly", MenuFrame, function()
    flyEnabled = not flyEnabled
    print("Bay / Fly:", flyEnabled)
end)

-- Bật/tắt menu
HubButton.MouseButton1Click:Connect(function()
    MenuFrame.Visible = not MenuFrame.Visible
end)

-- ================== Logic Fly ==================
RunService.RenderStepped:Connect(function()
    if flyEnabled then
        humanoid.PlatformStand = true
        local cam = workspace.CurrentCamera
        local moveDirection = Vector3.new()
        if game:GetService("UserInputService"):IsKeyDown(Enum.KeyCode.W) then moveDirection = moveDirection + cam.CFrame.LookVector end
        if game:GetService("UserInputService"):IsKeyDown(Enum.KeyCode.S) then moveDirection = moveDirection - cam.CFrame.LookVector end
        if game:GetService("UserInputService"):IsKeyDown(Enum.KeyCode.A) then moveDirection = moveDirection - cam.CFrame.RightVector end
        if game:GetService("UserInputService"):IsKeyDown(Enum.KeyCode.D) then moveDirection = moveDirection + cam.CFrame.RightVector end
        rootPart.CFrame = rootPart.CFrame + moveDirection.Unit * 1 -- tốc độ bay
    else
        humanoid.PlatformStand = false
    end
end)

-- ================== Logic Auto Farm Quái ==================
RunService.RenderStepped:Connect(function()
    if autoFarm then
        for _, npc in pairs(workspace:GetDescendants()) do
            if npc:IsA("Model") and npc:FindFirstChild("Humanoid") and npc ~= character then
                -- Local test: di chuyển gần NPC
                rootPart.CFrame = CFrame.new(npc.HumanoidRootPart.Position + Vector3.new(0,3,0))
                -- Local test: đánh quái (thêm animation hoặc damage nếu muốn)
                print("Đấm quái: "..npc.Name.." (local test)")
            end
        end
    end
end)

-- ================== Logic Auto Level ==================
local fakeLevel = 1
RunService.RenderStepped:Connect(function()
    if autoLevel then
        fakeLevel = fakeLevel + 0.1
        title.Text = "Level Local: "..math.floor(fakeLevel)
    end
end)
