local EventRemote = game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):FindFirstChild("EggEvent") or

_G.ForceESP = false

-- [[ HÀM CAN THIỆP SERVER - BẮT BUỘC HIỆN TRỨNG ]] --
function ForceRender()
    for _, v in pairs(game:GetDescendants()) do
        -- Truy quét theo ID hình dáng trứng indra, rocket (MeshId)
        if v:IsA("MeshPart") and (v.MeshId:find("1676678") or v.Size.X < 7) then
            if v:FindFirstChild("TouchInterest") or v.CanTouch == true then
                if not v:FindFirstChild("AdminESP") then
                    -- Bắt buộc hiện Highlight xanh rực
                    local hl = Instance.new("Highlight", v)
                    hl.Name = "AdminESP"
                    hl.FillColor = Color3.fromRGB(255, 0, 0)
                    hl.OutlineColor = Color3.fromRGB(255, 255, 255)
                    hl.AlwaysOnTop = true
                    
                    -- Hiện tên và tọa độ
                    local bg = Instance.new("BillboardGui", v)
                    bg.Name = "AdminText"
                    bg.AlwaysOnTop = true
                    bg.Size = UDim2.new(0, 200, 0, 50)
                    bg.StudsOffset = Vector3.new(0, 4, 0)
                    
                    local tl = Instance.new("TextLabel", bg)
                    tl.Size = UDim2.new(1, 0, 1, 0)
                    tl.BackgroundTransparency = 1
                    tl.Text = "🚨 EGG DETECTED 🚨"
                    tl.TextColor3 = Color3.fromRGB(255, 0, 0)
                    tl.TextScaled = true
                end
            end
        end
    end
end

})
