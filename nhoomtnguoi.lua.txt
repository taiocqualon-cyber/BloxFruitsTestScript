local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local TweenService = game:GetService("TweenService")
local UserInputService = game:GetService("UserInputService")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local rootPart = character:WaitForChild("HumanoidRootPart")
local humanoid = character:WaitForChild("Humanoid")

-- GUI
local ScreenGui = Instance.new("ScreenGui", player:WaitForChild("PlayerGui"))
local HubButton = Instance.new("TextButton", ScreenGui)
HubButton.Size = UDim2.new(0,100,0,50)
HubButton.Position = UDim2.new(0,20,0,100)
HubButton.Text = "VIP Hub"

local MenuFrame = Instance.new("Frame", ScreenGui)
MenuFrame.Size = UDim2.new(0,300,0,400)
MenuFrame.Position = UDim2.new(0,140,0,100)
MenuFrame.Visible = false

local title = Instance.new("TextLabel", MenuFrame)
title.Size = UDim2.new(1,0,0,50)
title.Text = "VIP Local Hub"
title.TextScaled = true
title.BackgroundTransparency = 1

-- Toggle biến
local autoFarm, flyEnabled = false, false
local isTweening = false

-- Nút toggle
HubButton.MouseButton1Click:Connect(function()
    MenuFrame.Visible = not MenuFrame.Visible
end)

-- Dummy buttons
local function createButton(name, callback)
    local btn = Instance.new("TextButton", MenuFrame)
    btn.Size = UDim2.new(1,-20,0,50)
    btn.Position = UDim2.new(0,10,0,50*(#MenuFrame:GetChildren()))
    btn.Text = name
    btn.MouseButton1Click:Connect(callback)
end

createButton("Auto Farm Quái", function() autoFarm = not autoFarm end)
createButton("Bay / Fly", function() flyEnabled = not flyEnabled humanoid.PlatformStand = flyEnabled end)

-- ================== Fly logic ==================
RunService.RenderStepped:Connect(function()
    if flyEnabled then
        local cam = workspace.CurrentCamera
        local moveDir = Vector3.new()
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDir = moveDir + cam.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDir = moveDir - cam.CFrame.LookVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDir = moveDir - cam.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDir = moveDir + cam.CFrame.RightVector end
        if moveDir.Magnitude > 0 then
            rootPart.CFrame = rootPart.CFrame + moveDir.Unit * 2
        end
    end
end)

-- ================== Auto Farm Tween ==================
local function moveToTween(targetPos)
    if isTweening then return end
    isTweening = true
    local tweenInfo = TweenInfo.new((rootPart.Position - targetPos).Magnitude / 20, Enum.EasingStyle.Linear)
    local tween = TweenService:Create(rootPart, tweenInfo, {CFrame = CFrame.new(targetPos + Vector3.new(0,3,0))})
    tween:Play()
    tween.Completed:Wait()
    isTweening = false
end

RunService.RenderStepped:Connect(function()
    if autoFarm and not isTweening then
        local closestNPC = nil
        local shortestDist = math.huge
        for _, npc in pairs(workspace:GetDescendants()) do
            if npc:IsA("Model") and npc:FindFirstChild("HumanoidRootPart") and npc ~= character then
                local dist = (npc.HumanoidRootPart.Position - rootPart.Position).Magnitude
                if dist < shortestDist then
                    shortestDist = dist
                    closestNPC = npc
                end
            end
        end
        if closestNPC then
            moveToTween(closestNPC.HumanoidRootPart.Position)
            print("Di chuyển tới quái:", closestNPC.Name)
        end
    end
end)
